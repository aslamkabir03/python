create database final
use final

--1
create table Employee(
EmpID int primary key,
EmpName varchar(100),
Dept varchar(100),
Salary int,
JoinDate date
);

create table Project(
ProjectID int,
ProjectName varchar(100),
EmpID int,
Budget int,
foreign key(EmpID) references Employee(EmpID)
);

--2
create proc report
as
begin
select EmpName,Dept,ProjectName,Budget from Employee inner join Project on Employee.EmpID=Project.EmpID
end

exec report

--3
create function EmpEx()
returns table
as
return(select EmpName,Dept,DATEDIFF(year,JoinDate,SYSDATETIME()) as Experience from Employee)

select * from dbo.EmpEx()

--4

update Employee
set Salary= case
when Salary<(select avg(Salary) from Employee) then Salary*1.1
else Salary*1.05
end

select * from Employee

--5
create view view_empinfo
as select EmpName,Dept, ProjectName from Employee inner join Project on Employee.EmpID=Project.EmpID

select * from view_empinfo

alter view view_empinfo
as select EmpName,Dept, ProjectName from Employee Employee inner join Project on Employee.EmpID=Project.EmpID where Dept='IT'

--6
create view report_2
with schemabinding
as select Dept,sum(isnull(Budget,0)) as Total_Budget,COUNT_BIG(*) as C from dbo.Employee inner join dbo.Project 
on Employee.EmpID=Project.EmpID group by Dept

select * from report_2

create unique clustered index idx_report_2
on report_2(Dept)
